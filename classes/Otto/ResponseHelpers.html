    def send_cookie name, value, ttl, secure=true
      secure = false if request.local?
      opts = {
        :value    => value, 
        :path     => '/', 
        :expires  => (Time.now + ttl + 10).utc,
        :secure   => secure
      }
      opts[:domain] = request.env['SERVER_NAME']
      #pp [:cookie, name, opts]
      set_cookie name, opts
    end
